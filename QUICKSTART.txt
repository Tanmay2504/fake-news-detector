"""
Quick Start Guide
"""

# 1. Install dependencies
pip install -r requirements.txt

# 2. Run setup (downloads NLTK data, checks models)
python setup.py

# 3. Add your trained models to models/ directory:
#    - models/random_forest.joblib
#    - models/lightgbm.joblib
#    - models/xgboost.joblib

# 4. (Optional) Add Google Cloud Vision credentials:
#    - google-vision-credentials.json

# 5. Start the server
python main.py

# Or use uvicorn directly:
uvicorn main:app --reload --host 0.0.0.0 --port 8000

# 6. Test the API
python test_api.py

# 7. Visit the API documentation
# http://localhost:8000/docs
